from application.db import db

class User(db.Model):
    __tablename__="user"
    id = db.Column(db.Integer, primary_key=True)
    firstname = db.Column(db.String(50))
    lastname = db.Column(db.String(50))
    username = db.Column(db.String(50), unique=True)
    password = db.Column(db.String(128))
    
class Beehive(db.Model):
    __tablename__="beehive"
    id = db.Column(db.Integer, primary_key=True)
    devicename = db.Column(db.String(50), unique=True)
    displayname = db.Column(db.String(50))
    temperature = db.Column(db.Float, default=0)
    humidity = db.Column(db.Float, default=0)
    weight = db.Column(db.Float, default=0)
    
class UserBeehive(db.Model):
    __tablename__="userbeehive"
    user_id = db.Column(db.Integer, db.ForeignKey('user.id', ondelete="CASCADE"), primary_key=True, nullable=False)
    user = db.relationship("user")
    device_id = db.Column(db.Integer, db.ForeignKey('beehive.id', ondelete="CASCADE"), primary_key=True, nullable=False)
    beehive = db.relationship("beehive")
    date = db.Column(db.DateTime, nullable=False)
